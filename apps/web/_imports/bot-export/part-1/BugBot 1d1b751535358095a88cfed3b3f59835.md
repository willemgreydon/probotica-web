# BugBot

### MVP Features

- 🧱 Check for broken links on a given WordPress site
- ⚙️ Capture basic JavaScript errors using Puppeteer
- 📬 Send bug summary via email using NodeMailer
- 📁 Use `.env` to store your domain and email credentials
- 🔁 Run from CLI or via a cronjob for constant scanning

## Step 3: `scanner.js`

```

const puppeteer = require('puppeteer');
const { URL } = require('url');

async function scanSite(targetUrl) {
  const browser = await puppeteer.launch({ headless: 'new' });
  const page = await browser.newPage();
  let errors = [];

  page.on('pageerror', (err) => {
    errors.push(`JavaScript Error: ${err.message}`);
  });

  await page.goto(targetUrl, { waitUntil: 'networkidle2' });

  const links = await page.$$eval('a', anchors =>
    anchors.map(a => a.href)
  );

  const brokenLinks = [];

  for (let link of links) {
    try {
      const res = await page.goto(link, { waitUntil: 'domcontentloaded', timeout: 10000 });
      if (!res || res.status() >= 400) {
        brokenLinks.push(`Broken link: ${link} → Status: ${res && res.status()}`);
      }
    } catch (err) {
      brokenLinks.push(`Failed link: ${link} → ${err.message}`);
    }
  }

  await browser.close();

  return { errors, brokenLinks };
}

module.exports = { scanSite };

```

## 📬 Step 4: `mailer.js`

```

const nodemailer = require('nodemailer');
require('dotenv').config();

async function sendReport({ subject, body }) {
  const transporter = nodemailer.createTransport({
    host: process.env.SMTP_HOST,
    port: parseInt(process.env.SMTP_PORT),
    secure: false,
    auth: {
      user: process.env.EMAIL_USER,
      pass: process.env.EMAIL_PASS,
    },
  });

  await transporter.sendMail({
    from: process.env.EMAIL_FROM,
    to: process.env.EMAIL_TO,
    subject,
    text: body,
  });
}

module.exports = { sendReport };

```

## 🧠 Step 5: `index.js`

```

require('dotenv').config();
const { scanSite } = require('./scanner');
const { sendReport } = require('./mailer');
const chalk = require('chalk');

(async () => {
  console.log(chalk.blue(`🔍 Scanning ${process.env.TARGET_URL} for bugs...`));

  const result = await scanSite(process.env.TARGET_URL);

  let report = '';

  if (result.errors.length > 0) {
    report += '⚠️ JavaScript Errors:\n' + result.errors.join('\n') + '\n\n';
  }

  if (result.brokenLinks.length > 0) {
    report += '🧱 Broken Links:\n' + result.brokenLinks.join('\n') + '\n';
  }

  if (report === '') {
    report = '✅ No issues detected.';
  }

  console.log(report);

  await sendReport({
    subject: `🚨 BugBot Report for ${process.env.TARGET_URL}`,
    body: report,
  });

  console.log(chalk.green('📬 Report sent successfully.'));
})();

```

## 🔁 Optional: Set up a Cronjob

On Linux/macOS:

```bash

crontab -e

```

Add:

```bash

*/30 * * * * /usr/bin/node /path/to/bugbot/index.js

```

---

## 🪟 On Windows

Use Task Scheduler:

- Action: Run `node.exe`
- Arguments: `C:\path\to\bugbot\index.js`
- Trigger: Every 30 minutes
